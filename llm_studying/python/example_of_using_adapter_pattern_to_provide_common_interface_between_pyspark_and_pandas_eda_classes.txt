Ah, got it — you have **existing classes** with their own methods, and you want to route to them based on the backend string. Let me show you a practical approach.

## The Adapter Pattern for Existing Classes

```python
class DataFrameAdapter:
    """Routes method calls to the appropriate backend classes."""
    
    def __init__(self, backend: str):
        if backend == "pandas":
            from mypackage.pandas_ops import (
                PandasStats, 
                PandasTransforms, 
                PandasAggregations
            )
            self.stats = PandasStats()
            self.transforms = PandasTransforms()
            self.aggregations = PandasAggregations()
            
        elif backend == "spark":
            from mypackage.spark_ops import (
                SparkStats, 
                SparkTransforms, 
                SparkAggregations
            )
            self.stats = SparkStats()
            self.transforms = SparkTransforms()
            self.aggregations = SparkAggregations()
            
        else:
            raise ValueError(f"Unknown backend: {backend}")
```

**Usage:**
```python
adapter = DataFrameAdapter("pandas")

# Call methods on your existing classes
adapter.stats.compute_mean(df, "column_a")
adapter.transforms.normalize(df)
adapter.aggregations.pivot_table(df, index="x", values="y")
```

## Alternative: Flatter Interface with `__getattr__`

If you want a single namespace that searches across all your classes:

```python
class DataFrameAdapter:
    """Provides a unified interface across multiple backend classes."""
    
    def __init__(self, backend: str):
        self._backend = backend
        self._instances = []
        
        if backend == "pandas":
            from mypackage.pandas_ops import PandasStats, PandasTransforms
            self._instances = [PandasStats(), PandasTransforms()]
            
        elif backend == "spark":
            from mypackage.spark_ops import SparkStats, SparkTransforms
            self._instances = [SparkStats(), SparkTransforms()]
        else:
            raise ValueError(f"Unknown backend: {backend}")
    
    def __getattr__(self, name):
        """Look up method across all registered class instances."""
        for instance in self._instances:
            if hasattr(instance, name):
                return getattr(instance, name)
        
        raise AttributeError(
            f"'{self._backend}' backend has no method '{name}'"
        )
```

**Usage:**
```python
adapter = DataFrameAdapter("spark")

# Methods are found automatically across your existing classes
adapter.compute_mean(df, "col")      # found in SparkStats
adapter.normalize(df)                 # found in SparkTransforms
```

## Configuration-Driven Approach

If you have many classes, a mapping dict keeps it cleaner:

```python
BACKEND_CLASSES = {
    "pandas": {
        "stats": "mypackage.pandas_ops.PandasStats",
        "transforms": "mypackage.pandas_ops.PandasTransforms",
        "aggregations": "mypackage.pandas_ops.PandasAggregations",
    },
    "spark": {
        "stats": "mypackage.spark_ops.SparkStats",
        "transforms": "mypackage.spark_ops.SparkTransforms",
        "aggregations": "mypackage.spark_ops.SparkAggregations",
    },
}

class DataFrameAdapter:
    def __init__(self, backend: str):
        if backend not in BACKEND_CLASSES:
            raise ValueError(f"Unknown backend: {backend}")
        
        for attr_name, class_path in BACKEND_CLASSES[backend].items():
            module_path, class_name = class_path.rsplit(".", 1)
            module = __import__(module_path, fromlist=[class_name])
            cls = getattr(module, class_name)
            setattr(self, attr_name, cls())
```
