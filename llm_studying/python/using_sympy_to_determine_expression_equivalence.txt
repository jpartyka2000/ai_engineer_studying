```python
import sympy as sp

# Define symbolic variables
made_qualified, count = sp.symbols('made_qualified count')

expr1 = -(sp.Sum(made_qualified) * 100.0) / sp.Count(count)

are_equal = sp.simplify(expr1 - expr2) == 0

# Print the result

```
