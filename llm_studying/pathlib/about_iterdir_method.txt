PATHLIB DEEP DIVE: iterdir()
═══════════════════════════════════════════════════════════════

BASIC SYNTAX:
───────────────────────────────────────────────────────────────
Path.iterdir()  # Returns a generator of path objects

WHAT DOES iterdir() DO?
───────────────────────────────────────────────────────────────
Yields Path objects for each item (files AND directories) in
the directory. It's the pathlib equivalent of os.listdir(),
but returns Path objects instead of strings.

BASIC USAGE:
───────────────────────────────────────────────────────────────
from pathlib import Path

p = Path("/home/user/projects")

# Iterate over all items
for item in p.iterdir():
    print(item)

# Output:
# PosixPath('/home/user/projects/app1')
# PosixPath('/home/user/projects/app2')
# PosixPath('/home/user/projects/notes.txt')
# PosixPath('/home/user/projects/.hidden')

KEY CHARACTERISTICS:
───────────────────────────────────────────────────────────────
1. Returns a GENERATOR (lazy evaluation, memory efficient)
2. Yields Path objects, not strings
3. Includes hidden files/directories (those starting with ".")
4. Does NOT recurse into subdirectories
5. Order is arbitrary (not sorted)
6. Only works on directories (raises NotADirectoryError on files)

COMPARISON WITH os.listdir():
───────────────────────────────────────────────────────────────
import os
from pathlib import Path

# os.listdir() returns strings
os.listdir("/home/user")  
# ['projects', 'documents', '.bashrc']

# iterdir() returns Path objects
list(Path("/home/user").iterdir())  
# [PosixPath('/home/user/projects'), 
#  PosixPath('/home/user/documents'),
#  PosixPath('/home/user/.bashrc')]

# With Path objects, you can immediately chain methods:
for item in Path("/home/user").iterdir():
    print(item.name, item.is_dir(), item.suffix)

COMMON PATTERNS:
───────────────────────────────────────────────────────────────
# Pattern 1: List only directories
dirs = [d for d in p.iterdir() if d.is_dir()]

# Pattern 2: List only files
files = [f for f in p.iterdir() if f.is_file()]

# Pattern 3: List only visible items (exclude hidden)
visible = [x for x in p.iterdir() if not x.name.startswith('.')]

# Pattern 4: Filter by extension
py_files = [f for f in p.iterdir() if f.suffix == '.py']

# Pattern 5: Get sorted listing
sorted_items = sorted(p.iterdir())  # Alphabetical
sorted_items = sorted(p.iterdir(), key=lambda x: x.name.lower())  # Case-insensitive

# Pattern 6: Sort directories first, then files
def dirs_first(path):
    return (not path.is_dir(), path.name.lower())
sorted_items = sorted(p.iterdir(), key=dirs_first)

# Pattern 7: Count items without loading all into memory
count = sum(1 for _ in p.iterdir())

# Pattern 8: Check if directory is empty
is_empty = not any(p.iterdir())

# Pattern 9: Get first item (or None if empty)
first = next(p.iterdir(), None)

# Pattern 10: Filter by multiple conditions
recent_py = [
    f for f in p.iterdir() 
    if f.is_file() 
    and f.suffix == '.py'
    and f.stat().st_mtime > some_timestamp
]

iterdir() vs glob() vs rglob():
───────────────────────────────────────────────────────────────
Method      | Recursive? | Pattern Matching? | Use Case
------------|------------|-------------------|------------------
iterdir()   | No         | No                | List all items
glob()      | Optional** | Yes               | Find by pattern
rglob()     | Yes        | Yes               | Recursive search

** glob() becomes recursive with "**" pattern

# These are roughly equivalent:
list(p.iterdir())           # All items
list(p.glob("*"))           # All items (using pattern)

# But glob() can filter:
list(p.glob("*.py"))        # Only .py files
list(p.glob("test_*"))      # Only items starting with test_

# And rglob() recurses:
list(p.rglob("*.py"))       # All .py files in all subdirectories

ERROR HANDLING:
───────────────────────────────────────────────────────────────
from pathlib import Path

# Error 1: Path doesn't exist
p = Path("/nonexistent")
list(p.iterdir())  # Raises FileNotFoundError

# Error 2: Path is a file, not a directory
p = Path("/home/user/file.txt")
list(p.iterdir())  # Raises NotADirectoryError

# Error 3: Permission denied
p = Path("/root")  # (if not root user)
list(p.iterdir())  # Raises PermissionError

# Safe pattern with error handling:
def safe_iterdir(path):
    p = Path(path)
    try:
        if not p.exists():
            print(f"Path does not exist: {p}")
            return []
        if not p.is_dir():
            print(f"Not a directory: {p}")
            return []
        return list(p.iterdir())
    except PermissionError:
        print(f"Permission denied: {p}")
        return []

PRACTICAL EXAMPLES:
───────────────────────────────────────────────────────────────
# Example 1: Simple directory listing with details
def list_directory(path):
    p = Path(path)
    for item in sorted(p.iterdir()):
        item_type = "DIR " if item.is_dir() else "FILE"
        size = item.stat().st_size if item.is_file() else "-"
        print(f"{item_type} {size:>10} {item.name}")

# Example 2: Find largest files in a directory
def find_largest_files(path, n=5):
    p = Path(path)
    files = [f for f in p.iterdir() if f.is_file()]
    return sorted(files, key=lambda f: f.stat().st_size, reverse=True)[:n]

# Example 3: Organize files by extension
def organize_by_extension(path):
    p = Path(path)
    organized = {}
    for item in p.iterdir():
        if item.is_file():
            ext = item.suffix or "no_extension"
            organized.setdefault(ext, []).append(item)
    return organized

# Example 4: Clean up empty directories
def remove_empty_dirs(path):
    p = Path(path)
    for item in p.iterdir():
        if item.is_dir() and not any(item.iterdir()):
            print(f"Removing empty directory: {item}")
            item.rmdir()

# Example 5: Recursive listing (manual implementation)
def recursive_list(path, indent=0):
    p = Path(path)
    for item in sorted(p.iterdir()):
        print("  " * indent + item.name)
        if item.is_dir():
            recursive_list(item, indent + 1)

REMEMBER - GENERATOR BEHAVIOR:
───────────────────────────────────────────────────────────────
# iterdir() returns a generator - it's exhaustible!
items = p.iterdir()
list1 = list(items)  # Contains all items
list2 = list(items)  # EMPTY! Generator exhausted

# If you need to iterate multiple times, convert to list first:
items = list(p.iterdir())
# Now you can use 'items' as many times as you want
