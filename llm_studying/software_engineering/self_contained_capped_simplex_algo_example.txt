# Define the projection function with detailed iteration logging
def project_to_capped_simplex_detailed(p: np.ndarray, S: float, lower: float = 0.0, upper: float = 1.0, iterations: int = 10):
    lo, hi = min(p) - upper, max(p) - lower
    print(f"Target sum S = {S}\n")
    print(f"{'Iter':>4} | {'delta(mid)':>10} | {'sum(clipped)':>12} | clipped values")
    print("-" * 60)
    for i in range(1, iterations + 1):
        mid = 0.5 * (lo + hi)
        clipped = np.clip(p - mid, lower, upper)
        total = clipped.sum()
        print(f"{i:4d} | {mid:10.4f} | {total:12.4f} | {clipped}")
        if total > S:
            lo = mid
        else:
            hi = mid
    delta = 0.5 * (lo + hi)
    final_clipped = np.clip(p - delta, lower, upper)
    print("\nFinal threshold delta ≈ {:.4f}".format(delta))
    print("Final projected vector:", final_clipped)

# Example data

project_to_capped_simplex_detailed(p, S, iterations=10)
