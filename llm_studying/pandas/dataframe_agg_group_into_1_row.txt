import pandas as pd

# Sample DataFrame
data = {'deal_id': [1, 1, 2, 2, 2, 3], 
        'amount_raised': [100, 200, 300, 400, 300, 500], 
        'col1': ['A', 'A', 'B', 'B', 'B', 'C'], 
        'col2': ['X', 'X', 'Y', 'Y', 'Y', 'Z']}
df = pd.DataFrame(data)

# Group by 'deal_id' and calculate the average for 'amount_raised'
# For other columns, we take the first value (since they are assumed to be the same for each group)
result = df.groupby('deal_id').agg({
    'amount_raised': 'mean',   # Average for 'amount_raised'
    'col1': 'first',           # Take the first value in 'col1' (assuming it's the same for all rows in a group)
    'col2': 'first'            # Take the first value in 'col2' (assuming it's the same for all rows in a group)
}).reset_index()

# Show the resulting DataFrame
print(result)
