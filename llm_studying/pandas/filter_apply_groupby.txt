import pandas as pd

# Sample DataFrame
data = {'deal_id': [1, 1, 2, 2, 2, 3], 
        'amount_raised': [100, 200, 300, 300, 300, 400], 
        'col1': ['A', 'A', 'B', 'B', 'B', 'C'], 
        'col2': ['X', 'X', 'Y', 'Y', 'Y', 'Z']}
df = pd.DataFrame(data)

# Get columns to compare, excluding 'deal_id' and 'amount_raised'
columns_to_check = df.columns.difference(['deal_id', 'amount_raised'])

# Group by 'deal_id' and filter groups with more than one row
groups = df.groupby('deal_id').filter(lambda x: len(x) > 1)

# Check if the values in each group are the same across the specified columns
different_values = groups.groupby('deal_id').apply(lambda x: (x[columns_to_check].nunique() > 1).any())

# Show deal_ids where values differ
print(different_values[different_values])
