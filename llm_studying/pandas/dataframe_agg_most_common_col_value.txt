import pandas as pd

# Sample DataFrame
data = {'deal_id': [1, 1, 2, 2, 2, 3], 
        'amount_raised': [100, 200, 300, 400, 300, 500], 
        'col1': ['A', 'A', 'B', 'B', 'C', 'C'], 
        'col2': ['X', 'X', 'Y', 'Y', 'Y', 'Z']}
df = pd.DataFrame(data)

# Group by 'deal_id' and calculate the average for 'amount_raised'
# For 'col1', we take the most common value (mode), and for other columns like 'col2', we use the first value
result = df.groupby('deal_id').agg({
    'amount_raised': 'mean',   # Average for 'amount_raised'
    'col1': lambda x: x.mode()[0],  # Take the most common value (mode) in 'col1'
    'col2': 'first'            # Take the first value in 'col2'
}).reset_index()

# Show the resulting DataFrame
print(result)
