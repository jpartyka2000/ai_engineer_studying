```python
import time
import openai

def query_gpt4o_with_retry(prompt, max_retries=5):
    retry_count = 0
    while retry_count < max_retries:
        try:
            response = openai.ChatCompletion.create(
                model="gpt-4o",
                messages=[{"role": "user", "content": prompt}],
                max_tokens=100
            )
            return response
        except openai.error.RateLimitError as e:
            print(f"Rate limit hit. Retrying in {2 ** retry_count} seconds...")
            time.sleep(2 ** retry_count)  # Exponential backoff
            retry_count += 1
    raise Exception("Exceeded max retries for GPT-4o request.")

# Example usage
prompt = "Write a summary of the GPT-4o rate limit error and how to handle it."
try:
    result = query_gpt4o_with_retry(prompt)
    print(result)
except Exception as e:
    print(f"Failed after retries: {e}")
```
